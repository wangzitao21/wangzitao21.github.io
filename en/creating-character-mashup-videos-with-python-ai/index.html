Editing Ideas#

  1. Collect the character materials for the mashup, such as making a mashup video of Monica Bellucci, you need to collect her movies, short films, etc. Put all the videos in the first folder named Material.

Do not include . in video file names. Because file extensions are modified based on ., including . in the file name will cause an error.

  1. Find a few photos of Monica Bellucci online and put them in the second folder named Portrait. These reference pictures will be used as references for identification from the video.

Note that the brightness of the reference pictures should not be too dark, as lower brightness can lead to slightly lower recognition accuracy. Do not use too many different sizes of images, as this can easily cause recognition errors. A simple solution is to directly take a few screenshots from existing videos.

  1. Split all the videos in the Material folder at their respective transitions, generating a large number of short clips, and put them in the Videos folder. This is done by the split function.
  2. Export the first frame of all short clips in the Videos folder as images and put them in the Images folder. This is done by the get_thumbnail_from_video function.
  3. Use facial recognition to compare each image in the Images folder with the reference images in the Portrait folder. Once they are identified as the same person, move the corresponding video from the Videos folder to the Output folder. This is done by the face_con function.
  4. Install the Beatedit plugin in Premiere Pro. Insert music and automatically match the rhythm with beat points.

import os
import shutil
import ffmpy
import face_recognition
import subprocess as sp

def split(materialPath, videosPath):
    """
    Split all materials at transition points and move them to the Videos folder
    """
    materialList = os.listdir(path=materialPath)
    for material in materialList:
        material = os.path.join(materialPath, material)
        cmd = "scenedetect --input %s detect-content list-scenes split-video --output %s" % (material, videosPath)
        print(cmd)
        p = sp.Popen(cmd, shell=True)
        p.wait()

def get_images_from_videos(videosPath, imagesPath):
    """
    Export the first frame of each video in the Videos folder as an image, and move it to the Images folder
    """
    videosList = os.listdir(path=videosPath) # Get all file names in the video folder, and store them in a list
    for video in videosList:
        image = video.replace(".mp4", ".png")
        ff = ffmpy.FFmpeg(
            inputs={os.path.join(videosPath, video): None},
            outputs={os.path.join(imagesPath, image): ['-ss', '00:0:01.000', '-vframes', '1']}
        )
        ff.run()

def facial_recognition(imagesPath, portraitPath, videosPath, outputPath):
    """
    Based on the image comparison results, move videos that indeed feature the same person to the Output folder
    """
    portraitsList = os.listdir(path=portraitPath)
    portraitEncoding = []
    for portrait in portraitsList:
        portrait = face_recognition.load_image_file(os.path.join(portraitPath, portrait))
        portraitEncoding.append(face_recognition.face_encodings(portrait)[0])
    
    imagesList = os.listdir(path=imagesPath)
    for image in imagesList:
	    imageName = image # for later use
        image = face_recognition.load_image_file(os.path.join(imagesPath, image))
        imageEncoding = face_recognition.face_encodings(image) # If there is no face information, an empty list is returned

        if len(imageEncoding) > 0:
            results = face_recognition.compare_faces(portraitEncoding, imageEncoding[0]) # portraitEncoding is a list, can pass multiple images to compare
            if results[0] == True:
                print("The image contains the target person")
                video = imageName.split(".")[0] + ".mp4"
                output = shutil.move(os.path.join(videosPath, video), outputPath)
                print(f"Moved {video} to: {output}")
            else:
                print("The image does not contain the target person")

def create_folder(foldersList):
    """
    Check if the folder exists, if not, create a new one
    """
    for folder in foldersList:
        if not os.path.exists(folder):
            os.mkdir(folder)

if __name__ == '__main__':
    materialPath = r"./Material"
    videosPath = r"./Videos"
    imagesPath = r"./Images"
    portraitPath = r"./Portrait"
    outputPath = r"./Output"

    create_folder([videosPath, imagesPath, outputPath])
    split(materialPath=materialPath, videosPath=videosPath)
    get_images_from_videos(videosPath=videosPath, imagesPath=imagesPath)
    facial_recognition(videosPath=videosPath, imagesPath=imagesPath, portraitPath=portraitPath, outputPath=outputPath)
